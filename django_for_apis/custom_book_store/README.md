import sqlite3
con=sqlite3.connect("db.sqlite3")
def dict_factory(cursor, row):
   d = {}
   for idx, col in enumerate(cursor.description):
       d[col[0]] = row[idx]
   return d


con.row_factory = dict_factory
cur=con.cursor()
cur.execute("select * from sqlite_master where type='table'")
